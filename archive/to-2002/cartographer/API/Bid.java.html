public class Bid implements java.io.Serializable {
    
    private int amount;
    private int id;

    public Bid(int id, int amount) {
	this.amount = amount;
	this.id = id;
    }
    
    public int getAmount(){
	return amount;
    }
    
    public int getID(){
	return id;
    }
    
} // Bid
