package uk.ac.ic.doc.ise3.sp;
import java.io.*;

class Transaction  
{
    private int amount,person;
    private long time;
    private String comment;
    
    public int getAmount() { return amount; }
    
    public Transaction(int a, int p, String c)
    {
		time = BankImpl.s_getTime();
		comment = c;
		amount = a;
		person = p;
    }
    
    public int printStatementLine(int runningTotal,PrintStream psout)
    {
		psout.println("£" + runningTotal + "\t£" + amount + "\t" + person + "\t" + time + "\t" + comment);
		return runningTotal + amount;
    }
}
