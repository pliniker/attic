package uk.ac.ic.doc.ise3.sp;

/**
 * Lot.java
 *
 *@author Gus Paul
 *@date 20/6/2001
 *
 *This class represents the object that the server creates and passes to the cartographers, after a 
 *region has been submitted for auction, allowing the cartographers to make a decision on what to bid
 *for an object.  
 *
 *This object is saved on the server after the auction, and the various get Lot methods will return an 
 *object like this with all the bid variables filled after the auction takes place.  This is so you can
 *base your strategy on what other people are doing.
 *
 *
 *
 */ 

import java.util.Vector;

public class Lot implements java.io.Serializable {

    private Region region;
    private int lotNo;
    private int accuracy, reserve;
    private Vector bids = new Vector();

    public Lot(Region region, int no, int accuracy){
	this.region = region;	 
	lotNo = no;
	this.accuracy = accuracy;
	this.reserve = 0;
    }
    
    public int getLotNo(){
	return lotNo;
    }
    
    public Region getRegion(){
	return region;
    }
    
    public int getAccuracy(){
	return accuracy;
    }
    
    public int getReserve(){
	//will only not be 0 AFTER auction
	return reserve;
    }
    
    public Bid getBid(int n) throws Exception{
	return (Bid)bids.elementAt(n);
    }

    public Vector getAllBids(){
	return bids;
    }

    void setBid(Bid bid){
	bids.addElement(bid);
    }
  
    void setReserve(int amount){
	reserve = amount;
    }
    
    void setRegion(Region r){
	region = r;
    }

}
