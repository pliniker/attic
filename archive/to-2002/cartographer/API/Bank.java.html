package uk.ac.ic.doc.ise3.sp;
import java.util.Vector;

public interface Bank extends java.rmi.Remote 
{
	// Group Types for use in openAccount
	public static final int   CARTOGRAPHER     = 4096;
	public static final int   ROBOT            = 2048;
	public static final int   SERVICEPROVIDER  = 1024;
	
        // Some constants
	public static final String SERVERPATH            = "rmi://texel25.doc.ic.ac.uk:5001";
	public static final int    TIMEPENALTY           = 10;
	public static final int    ACCURACYPENALTY       = 10;
	public static final int    OVERDRAFTLIMIT        = -10000;
	public static final int    OVERDRAWNFINE         = 10;
	public static final int    INTERESTTIMEBASE      = 5;
	public static final float  INTERESTRATE          = (float)0.01;
 	public static final float  OVERDRAWNINTERESTRATE = (float)0.01;
 
 	// Open new account and register for all other services
	// returns your account number for use in getBalance
	// and in Auction API
	public int openAccount(String Name, int groupType, int groupNumber) 
			throws java.rmi.RemoteException;
	
	// Get current balance of your account
	public int getBalance(int accountNum) 
			throws java.rmi.RemoteException, InvalidAccountNoException;
	
        // Gets the average difference in time taken from time predicted for
	// a robot with RobotNo to map a given area
	public long  getTimeReputation(int RobotNo) 
			throws java.rmi.RemoteException, InvalidAccountNoException;

	// Gets the average accuracy
	// a robot with RobotNo to map a given area
	public int   getAccuracyReputation(int RobotNo) 
			throws java.rmi.RemoteException, InvalidAccountNoException;

	// Ignore this, for internal use only
	public int   getNextRegionID() throws java.rmi.RemoteException;
	
	// Returns a vector of all cartographers in system
	public Vector getCartographers() throws java.rmi.RemoteException;
	

}
