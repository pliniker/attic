package uk.ac.ic.doc.ise3.sp;

/**
 * Quote.java
 *
 *@author Gus Paul
 *@date 7/7/2001
 *
 *This class represents the object that a robot creates and passes to the server, allowing
 *the cartographer who offered the job to make a decision on which quote to accept.  
 *
 */ 

import uk.ac.ic.doc.ise3.sp.*;

public class Quote implements java.io.Serializable{

    private int robotID, price, accuracy;
    private long time;
    private Region region;
    
    public Quote(int robotID, Region region, int price, long time, int accuracy){
	this.robotID = robotID;
	this.region= region;
	this.price = price;
	this.time = time;
	this.accuracy = accuracy;
    } 
    
    public int getRobotID(){
	return robotID;
    }

    public Region getRegion(){
	return region;
    }
    
    public int getPrice(){
	return price;
    }
    
    public long getTime(){
	return time;
    }
    
    public int getAccuracy(){
	return accuracy;
    }

}
